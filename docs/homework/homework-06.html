                  re_formula = NA) %>% 
  rowwise() %>% 
  mutate(interaction = paste0(pred,"+", size)) %>%

  stat_halfeye(aes(y = interaction, x = .epred/density,)) + 
  theme_classic() + 
  labs(y = "", x = "Survival Rate")
